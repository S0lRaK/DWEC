                    <button onclick="sumaVariables()">Suma</button>
                    <p id="variablesLocals"></p>
                    <script>
                        function sumaVariables() {
                            var x = 3;
                            var y = 1;
                            var resultat = x + y;
                            document.getElementById("variablesLocals").innerHTML = x + " + " + y + " = " + resultat;
                        }
                    </script>
                    <p>Clica el botó per a realitzar l'operació amb variables GLOBALS</p>
                    <button onclick="restaVariables()">Resta</button>
                    <p id="variablesGlobals"></p>
                    <script>
                        var z = 4, w = 2;
                        function restaVariables()
                        {
                            var resultat = z - w;
                            document.getElementById("variablesGlobals").innerHTML = z + " - " + w + " = " + resultat;
                        }
                    </script>
